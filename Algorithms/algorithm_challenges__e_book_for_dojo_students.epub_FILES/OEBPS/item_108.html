ShiftArrayValsLeft(arr)

Given an array, shift values leftward by one. Drop first values and leave extra '0' value(s) at end.

 

function shiftArrValsLeft(arr) {

for (var idx = 1;idx < arr.length;idx++) {

arr[idx + 1] = arr[idx];

}

arr[arr.length - 1] = 0;

return arr;

}
